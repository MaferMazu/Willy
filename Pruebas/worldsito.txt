begin-task prog1 on pathworld
    drop manzana;

    define nextmove as begin
        if front-clear then move;
        else if right-clear then begin
            turn-right;
            move;
        end;
        else if left-clear then begin
            turn-left;
            move;
        end;
    end;

    nextmove;
    while not found(final) do begin
        -- visitado
        if not found(manzana) then drop manzana;
        else
          terminate;

        if looking-north then nextmove;
        else if looking-south then nextmove;
        else if looking-east then nextmove;
        else nextmove;
    end;
end-task

begin-task prog3 on pathworld
    define nextmove as begin
        if found(final) or found(manzana) then terminate;
        drop manzana;

        define getmove as begin
            if front-clear then move;
            else if right-clear then begin
                turn-right;
                move;
            end;
            else if left-clear then begin
                turn-left;
                move;
            end;
        end;
        getmove;
        nextmove;
    end;

    nextmove;
end-task
