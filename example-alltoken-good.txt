{{Definiendo un programa}}
begin-work on world1
begin
repeat 2 times move;
if willy-happy then move;
define L as turn-left;
define R as turn-right;
define M as move;
repeat 2 times drop table;
while willy-can-learn do drop mirror;
R;
if willy-can-learn then L else L;
repeat 3 times M;
R;
M;
if found flower then pick flower;
clear willy-happy;
set willy-happy to true;
flip willy-can-learn;
if carrying flower then drop flower;
R;
M;
if front-clear and left-clear an right-clear and looking-north then flip looking-north;
if looking-east or not looking-south then set looking-west;
end
end-work
terminate
