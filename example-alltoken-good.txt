{{Definiendo un programa}}
begin-task tareita on world1
begin
repeat 2 times move;
if willyhappy then move;
define L as turn-left;
define R as turn-right;
define M as move;
repeat 2 times drop table;
while willycanlearn do drop mirror;
R;
if willycanlearn then L else L;
repeat 3 times M;
R;
M;
if found (flower) then pick flower;
clear willyhappy;
set willyhappy to true;
flip willycanlearn;
if carrying (flower) then drop flower;
R;
M;
terminate;
if front-clear and left-clear and right-clear and looking-north then flip looking-north;
if looking-east or not looking-south then set looking-west;
end
end-task
